<CardGroup cols={3}>
  <Card title="VMSWO Assessments" icon="list-check">

  </Card>
  <Card title="Conjugate Reinforcement" icon="hand-fist">
    Force-based behavioral engagement measurement
  </Card>
  <Card title="Correspondence Analysis" icon="chart-scatter">
